8/// A combatant in an encounter
9#[derive(Debug, Clone, Serialize, Deserialize)]
10pub struct Combatant {
11    pub name: String,
12    pub level: u8,
13    pub class: Class,
14    pub ancestry: Ancestry,
15    pub attributes: Attributes,
16    pub hp: HitPoints,
17    pub stress: Stress,
18    pub evasion: u8,
19    pub armor: u8,
20    pub initiative: u8,
21    pub is_player: bool,
22}

24impl Combatant {
25    /// Save combatant to a JSON file
26    ///
27    /// # Examples
28    ///
29    /// ```no_run
30    /// use daggerheart_engine::combat::simulation::Combatant;
31    /// use daggerheart_engine::character::{Class, Ancestry, Attributes};
32    ///
33    /// let warrior = Combatant::player(
34    ///     "Grom",
35    ///     1,
36    ///     Class::Warrior,
37    ///     Ancestry::Orc,
38    ///     Attributes::from_array([2, 1, 1, 0, 0, -1]).unwrap(),
39    /// );
40    ///
41    /// warrior.save_to_file("grom.json").unwrap();
42    /// ```
43    pub fn save_to_file(&self, path: &str) -> Result<(), std::io::Error> {
44        let json = serde_json::to_string_pretty(self)
45            .map_err(|e| std::io::Error::new(std::io::ErrorKind::InvalidData, e))?;
46        std::fs::write(path, json)?;
47        Ok(())
48    }

50    /// Load combatant from a JSON file
51    ///
52    /// # Examples
53    ///
54    /// ```no_run
55    /// use daggerheart_engine::combat::simulation::Combatant;
56    ///
57    /// let warrior = Combatant::load_from_file("grom.json").unwrap();
58    /// println!("Loaded: {}", warrior.name);
59    /// ```
60    pub fn load_from_file(path: &str) -> Result<Self, std::io::Error> {
61        let json = std::fs::read_to_string(path)?;
62        serde_json::from_str(&json)
63            .map_err(|e| std::io::Error::new(std::io::ErrorKind::InvalidData, e))
64    }

66    /// Create a new player character combatant
67    ///
68    /// # Examples
69    ///
70    /// ```
71    /// use daggerheart_engine::combat::simulation::Combatant;
72    /// use daggerheart_engine::character::{Class, Ancestry, Attributes};
73    ///
74    /// let warrior = Combatant::player(
75    ///     "Grom",
76    ///     1,
77    ///     Class::Warrior,
78    ///     Ancestry::Orc,
79    ///     Attributes::from_array([2, 1, 1, 0, 0, -1]).unwrap(),
80    /// );
81    ///
82    /// assert_eq!(warrior.name, "Grom");
83    /// assert!(warrior.is_player);
84    /// ```
85    pub fn player(
86        name: impl Into<String>,
87        level: u8,
88        class: Class,
89        ancestry: Ancestry,
90        attributes: Attributes,
91    ) -> Self {
92        let base_hp = class.starting_hp();
93        let hp_total = (base_hp as i16 + ancestry.hp_modifier() as i16).max(1) as u8;
94
95        let base_evasion = class.starting_evasion();
96        let evasion_total = (base_evasion as i16 + ancestry.evasion_modifier() as i16).max(1) as u8;
97
98        Self {
99            name: name.into(),
100            level,
101            class,
102            ancestry,
103            attributes,
104            hp: HitPoints::new(hp_total),
105            stress: Stress::new(),
106            evasion: evasion_total,
107            armor: 0, // Can be set later with equipment
108            initiative: 0,
109            is_player: true,
110        }
111    }
112
113    /// Create a new enemy combatant
114    ///
115    /// # Examples
116    ///
117    /// ```
118    /// use daggerheart_engine::combat::simulation::Combatant;
119    ///
120    /// let goblin = Combatant::enemy("Goblin Scout", 1, 4, 13, 1);
121    ///
122    /// assert_eq!(goblin.name, "Goblin Scout");
123    /// assert!(!goblin.is_player);
124    /// ```
125    pub fn enemy(name: impl Into<String>, level: u8, hp: u8, evasion: u8, armor: u8) -> Self {
126        // For enemies, use valid placeholder stats
127        Self {
128            name: name.into(),
129            level,
130            class: Class::Rogue, // Placeholder - enemies don't need real classes
131            ancestry: Ancestry::Goblin, // Placeholder
132            attributes: Attributes::from_array([2, 1, 1, 0, 0, -1]).unwrap(),
133            hp: HitPoints::new(hp),
134            stress: Stress::new(),
135            evasion,
136            armor,
137            initiative: 0,
138            is_player: false,
139        }
140    }
141
142    /// Set armor value
143    pub fn with_armor(mut self, armor: u8) -> Self {
144        self.armor = armor;
145        self
146    }
147
148    /// Roll initiative
149    pub fn roll_initiative(&mut self) {
150        let roll = DualityRoll::roll();
151        let result = roll.with_modifier(0); // Could add Agility modifier
152        self.initiative = (result.total % 20) as u8; // Cap at 20
153    }
154
155    /// Check if combatant is alive
156    pub fn is_alive(&self) -> bool {
157        self.hp.is_alive()
158    }
159
160    /// Take damage
161    pub fn take_damage(&mut self, amount: u8) {
162        self.hp.take_damage(amount);
163    }
164
165    /// Gain stress
166    pub fn gain_stress(&mut self, amount: u8) {
167        self.stress.gain(amount);
168    }
169}

171/// Combat encounter state
172#[derive(Debug, Clone, Serialize, Deserialize)]
173pub struct CombatEncounter {
174    pub combatants: Vec<Combatant>,
175    pub round: u32,
176    pub turn_order: Vec<usize>,
177    pub current_turn: usize,
178    pub hope: Hope,
179    pub fear: Fear,
180}
181
182impl CombatEncounter {
183    /// Save combat encounter to a JSON file
184    ///
185    /// # Examples
186    ///
187    /// ```no_run
188    /// use daggerheart_engine::combat::simulation::CombatEncounter;
189    ///
190    /// let encounter = CombatEncounter::new(5);
191    /// encounter.save_session("encounter.json").unwrap();
192    /// ```
193    pub fn save_session(&self, path: &str) -> Result<(), std::io::Error> {
194        let json = serde_json::to_string_pretty(self)
195            .map_err(|e| std::io::Error::new(std::io::ErrorKind::InvalidData, e))?;
196        std::fs::write(path, json)?;
197        Ok(())
198    }

200    /// Load combat encounter from a JSON file
201    ///
202    /// # Examples
203    ///
204    /// ```no_run
205    /// use daggerheart_engine::combat::simulation::CombatEncounter;
206    ///
207    /// let encounter = CombatEncounter::load_session("encounter.json").unwrap();
208    /// println!("Round: {}", encounter.round);
209    /// ```
210    pub fn load_session(path: &str) -> Result<Self, std::io::Error> {
211        let json = std::fs::read_to_string(path)?;
212        serde_json::from_str(&json)
213            .map_err(|e| std::io::Error::new(std::io::ErrorKind::InvalidData, e))

216    /// Create a new combat encounter
217    ///
218    /// # Examples
219    ///
220    /// ```
221    /// use daggerheart_engine::combat::simulation::CombatEncounter;
222    ///
223    /// let encounter = CombatEncounter::new(5); // Party Hope max = 5
224    /// assert_eq!(encounter.round, 0);
225    /// assert_eq!(encounter.combatants.len(), 0);
226    /// ```
227    pub fn new(hope_max: u8) -> Self {
228        Self {
229            combatants: Vec::new(),
230            round: 0,
231            turn_order: Vec::new(),
232            current_turn: 0,
233            hope: Hope::new(hope_max),
234            fear: Fear::new(),
235        }
236    }
237
238    /// Add a combatant to the encounter
239    ///
240    /// # Examples
241    ///
242    /// ```
243    /// use daggerheart_engine::combat::simulation::{CombatEncounter, Combatant};
244    /// use daggerheart_engine::character::{Class, Ancestry, Attributes};
245    ///
246    /// let mut encounter = CombatEncounter::new(5);
247    /// let warrior = Combatant::player(
248    ///     "Grom",
249    ///     1,
250    ///     Class::Warrior,
251    ///     Ancestry::Orc,
252    ///     Attributes::from_array([2, 1, 1, 0, 0, -1]).unwrap(),
253    /// );
254    ///
255    /// encounter.add_combatant(warrior);
256    /// assert_eq!(encounter.combatants.len(), 1);
257    /// ```
258    pub fn add_combatant(&mut self, combatant: Combatant) {
259        self.combatants.push(combatant);
260    }
261
262    /// Start the encounter (roll initiative for all combatants)
263    pub fn start(&mut self) {
264        // Roll initiative for all combatants
265        for combatant in &mut self.combatants {
266            combatant.roll_initiative();
267        }
268
269        // Sort by initiative (highest first)
270        let mut indices: Vec<usize> = (0..self.combatants.len()).collect();
271        indices.sort_by(|&a, &b| {
272            self.combatants[b]
273                .initiative
274                .cmp(&self.combatants[a].initiative)
275        });
276        self.turn_order = indices;
277
278        // Start round 1
279        self.round = 1;
280        self.current_turn = 0;
281    }
282
283    /// Get the current combatant's index
284    pub fn current_combatant_index(&self) -> Option<usize> {
285        if self.current_turn < self.turn_order.len() {
286            Some(self.turn_order[self.current_turn])
287        } else {
288            None
289        }
290    }
291
292    /// Get the current combatant
293    pub fn current_combatant(&self) -> Option<&Combatant> {
294        self.current_combatant_index()
295            .and_then(|idx| self.combatants.get(idx))
296    }
297
298    /// Get a mutable reference to the current combatant
299    pub fn current_combatant_mut(&mut self) -> Option<&mut Combatant> {
300        self.current_combatant_index()
301            .and_then(|idx| self.combatants.get_mut(idx))
302    }
303
304    /// Advance to the next turn
305    pub fn next_turn(&mut self) {
306        self.current_turn += 1;
307
308        // If we've gone through all combatants, start new round
309        if self.current_turn >= self.turn_order.len() {
310            self.round += 1;
311            self.current_turn = 0;
312
313            // Remove dead combatants from turn order
314            self.turn_order
315                .retain(|&idx| self.combatants[idx].is_alive());
316        }
317    }
318
319    /// Check if combat is over
320    pub fn is_over(&self) -> bool {
321        // Combat is over if all players are dead or all enemies are dead
322        let players_alive = self.combatants.iter().any(|c| c.is_player && c.is_alive());
323        let enemies_alive = self.combatants.iter().any(|c| !c.is_player && c.is_alive());
324
325        !players_alive || !enemies_alive
326    }
327
328    /// Get victory status (true = players won, false = enemies won)
329    pub fn player_victory(&self) -> Option<bool> {
330        if !self.is_over() {
331            return None;
332        }
333
334        let players_alive = self.combatants.iter().any(|c| c.is_player && c.is_alive());
335
336        Some(players_alive)
337    }
338}
