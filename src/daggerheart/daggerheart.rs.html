1//! Daggerheart CLI - Command-line interface for the Daggerheart engine
2//!
3//! This CLI lets you:
4//! - Create and manage characters
5//! - Roll dice (basic, duality, damage)
6//! - Run combat simulations
7//! - Save and load game state

184fn main() {
185    let cli = Cli::parse();
186
187    let result = match cli.command {
188        Commands::Char(cmd) => handle_char_command(cmd),
189        Commands::Roll(cmd) => handle_roll_command(cmd),
190        Commands::Combat(cmd) => handle_combat_command(cmd),
191        Commands::Classes => {
192            list_classes();
193            Ok(())
194        }
195        Commands::Ancestries => {
196            list_ancestries();
197            Ok(())
198        }
199    };
200
201    if let Err(e) = result {
202        eprintln!("‚ùå Error: {}", e);
203        std::process::exit(1);
204    }
205}

207fn handle_char_command(cmd: CharCommands) -> Result<(), Box<dyn std::error::Error>> {
208    match cmd {
209        CharCommands::Create {
210            name,
211            class,
212            ancestry,
213            level,
214            output,
215            attributes,
216        } => {
217            // Parse class
218            let class = parse_class(&class)?;
219
220            // Parse ancestry
221            let ancestry = parse_ancestry(&ancestry)?;
222
223            // Parse attributes
224            let attr_values: Vec<i8> = attributes
225                .split(',')
226                .map(|s| s.trim().parse())
227                .collect::<Result<Vec<_>, _>>()?;
228
229            if attr_values.len() != 6 {
230                return Err("Attributes must have exactly 6 values".into());
231            }
232
233            let mut attr_array = [0i8; 6];
234            attr_array.copy_from_slice(&attr_values);
235            let attributes = Attributes::from_array(attr_array)?;
236
237            // Create character
238            let character = Combatant::player(name.clone(), level, class, ancestry, attributes);
239
240            // Create initial progress
241            let progress = CharacterProgress::new();
242
243            // Save files
244            let char_file = output.unwrap_or_else(|| format!("{}_char.json", name));
245            let progress_file = format!("{}_progress.json", name);
246
247            character.save_to_file(&char_file)?;
248            progress.save_to_file(&progress_file)?;
249
250            println!("‚úÖ Character created!");
251            println!("  Name: {}", character.name);
252            println!("  Class: {}", character.class);
253            println!("  Ancestry: {}", character.ancestry);
254            println!("  Level: {}", character.level);
255            println!("  HP: {}/{}", character.hp.current, character.hp.maximum);
256            println!("  Evasion: {}", character.evasion);
257            println!("\nüìÅ Files saved:");
258            println!("  Character: {}", char_file);
259            println!("  Progress: {}", progress_file);
260        }
261
262        CharCommands::Show { file } => {
263            let character = Combatant::load_from_file(&file)?;
264
265            println!("=== {} ===", character.name);
266            println!("  Class: {}", character.class);
267            println!("  Ancestry: {}", character.ancestry);
268            println!("  Level: {}", character.level);
269            println!();
270            println!("  HP: {}/{}", character.hp.current, character.hp.maximum);
271            println!("  Stress: {}/5", character.stress.current);
272            println!("  Evasion: {}", character.evasion);
273            println!("  Armor: {}", character.armor);
274            println!();
275            println!("Attributes:");
276            println!("  Agility:   {:+}", character.attributes.agility);
277            println!("  Strength:  {:+}", character.attributes.strength);
278            println!("  Finesse:   {:+}", character.attributes.finesse);
279            println!("  Instinct:  {:+}", character.attributes.instinct);
280            println!("  Presence:  {:+}", character.attributes.presence);
281            println!("  Knowledge: {:+}", character.attributes.knowledge);
282
283            // Try to load progress file
284            let progress_file = file.replace("_char.json", "_progress.json");
285            if let Ok(progress) = CharacterProgress::load_from_file(&progress_file) {
286                println!();
287                println!("Progress:");
288                println!("  Level: {}", progress.level);
289                println!(
290                    "  XP: {} / {}",
291                    progress.experience,
292                    progress.xp_for_next_level()
293                );
294                println!("  Cards: {:?}", progress.available_cards);
295            }
296        }
297
298        CharCommands::LevelUp { file, card } => {
299            let mut progress = CharacterProgress::load_from_file(&file)?;
300
301            if !progress.can_level_up() {
302                println!(
303                    "‚ùå Not enough XP to level up (need {}, have {})",
304                    progress.xp_for_next_level(),
305                    progress.experience
306                );
307                return Ok(());
308            }
309
310            let old_level = progress.level;
311            progress.level_up()?;
312
313            println!("üéâ LEVEL UP!");
314            println!("  {} ‚Üí {}", old_level, progress.level);
315            println!("  Remaining XP: {}", progress.experience);
316
317            if let Some(card_id) = card {
318                progress.add_card(&card_id);
319                println!("  ‚ú® Learned: {}", card_id);
320            }
321
322            progress.save_to_file(&file)?;
323            println!("\n‚úÖ Progress saved to {}", file);
324        }
325
326        CharCommands::AddXp { file, amount } => {
327            let mut progress = CharacterProgress::load_from_file(&file)?;
328
329            progress.add_experience(amount);
330            println!("üìà Added {} XP", amount);
331            println!("  Total XP: {}", progress.experience);
332            println!("  Level: {}", progress.level);
333            println!("  XP for next level: {}", progress.xp_for_next_level());
334
335            if progress.can_level_up() {
336                println!("\nüí° You can now level up! Run:");
337                println!("   daggerheart char level-up {}", file);
338            }
339
340            progress.save_to_file(&file)?;
341            println!("\n‚úÖ Progress saved");
342        }
343    }
344
345    Ok(())
346}

348fn handle_roll_command(cmd: RollCommands) -> Result<(), Box<dyn std::error::Error>> {
349    match cmd {
350        RollCommands::Die { die, count } => {
351            let die_type = parse_die(&die)?;
352
353            println!("üé≤ Rolling {}x{}:", count, die);
354            let mut total = 0;
355            let mut rolls = Vec::new();
356
357            for i in 1..=count {
358                let roll = die_type.roll();
359                total += roll;
360                rolls.push(roll);
361
362                if count <= 10 {
363                    println!("  Roll {}: {}", i, roll);
364                }
365            }
366
367            if count > 1 {
368                println!("\nRolls: {:?}", rolls);
369                println!("Total: {}", total);
370                println!("Average: {:.2}", total as f64 / count as f64);
371            }
372        }
373
374        RollCommands::Duality {
375            modifier,
376            advantage,
377        } => {
378            let roll = DualityRoll::roll();
379            let result = if advantage {
380                // Combine advantage and modifier manually
381                let adv_result = roll.with_advantage();
382                let total = (adv_result.total as i16 + modifier as i16) as u16;
383                DualityResult {
384                    roll: adv_result.roll,
385                    modifier,
386                    advantage_die: adv_result.advantage_die,
387                    total,
388                    controlling: adv_result.controlling,
389                    is_critical: adv_result.is_critical,
390                }
391            } else {
392                roll.with_modifier(modifier)
393            };
394
395            println!("üé≤ Duality Roll:");
396            println!("  Hope die: {}", roll.hope);
397            println!("  Fear die: {}", roll.fear);
398
399            if let Some(d6) = result.advantage_die {
400                println!("  Advantage die: {}", d6);
401            }
402
403            if modifier != 0 {
404                println!("  Modifier: {:+}", modifier);
405            }
406
407            println!("  Total: {}", result.total);
408
409            if result.is_critical {
410                println!("\nüåü CRITICAL! (Doubles: {})", roll.hope);
411            }
412
413            // Determine success type (assuming difficulty is met)
414            match result.controlling {
415                ControllingDie::Hope => println!("\n‚úÖ Hope controls! üåü"),
416                ControllingDie::Fear => println!("\n‚ö†Ô∏è Fear controls... üíÄ"),
417                ControllingDie::Tied => println!("\nüîÑ Tied"),
418            }
419        }
420
421        RollCommands::Damage { dice } => {
422            // Parse damage dice string (e.g., "2d6+3", "1d8+1d4+2")
423            let damage_dice = parse_damage_dice(&dice)?;
424            let result = damage_dice.roll();
425
426            println!("üé≤ Damage Roll: {}", dice);
427            println!("  Individual rolls: {:?}", result.rolls);
428            println!("  Total damage: {}", result.total);
429        }
430    }
431
432    Ok(())
433}

435fn handle_combat_command(cmd: CombatCommands) -> Result<(), Box<dyn std::error::Error>> {
436    match cmd {
437        CombatCommands::New { hope, output } => {
438            let encounter = CombatEncounter::new(hope);
439            encounter.save_session(&output)?;
440
441            println!("‚úÖ Combat encounter created!");
442            println!("  Hope pool: {}", hope);
443            println!("  File: {}", output);
444            println!("\nüí° Add combatants with:");
445            println!("   daggerheart combat add {} --character <file>", output);
446            println!("   daggerheart combat add {} --enemy <name>", output);
447        }
448
449        CombatCommands::Add {
450            file,
451            character,
452            enemy,
453            level,
454            hp,
455            evasion,
456            armor,
457        } => {
458            let mut encounter = CombatEncounter::load_session(&file)?;
459
460            if let Some(char_file) = character {
461                let combatant = Combatant::load_from_file(&char_file)?;
462                println!("‚ûï Adding player: {}", combatant.name);
463                encounter.add_combatant(combatant);
464            } else if let Some(enemy_name) = enemy {
465                let combatant = Combatant::enemy(enemy_name.clone(), level, hp, evasion, armor);
466                println!("‚ûï Adding enemy: {}", enemy_name);
467                println!("  HP: {}, Evasion: {}, Armor: {}", hp, evasion, armor);
468                encounter.add_combatant(combatant);
469            } else {
470                return Err("Must specify either --character or --enemy".into());
471            }
472
473            encounter.save_session(&file)?;
474            println!(
475                "\n‚úÖ Combatant added! Total: {}",
476                encounter.combatants.len()
477            );
478        }
479
480        CombatCommands::Start { file } => {
481            let mut encounter = CombatEncounter::load_session(&file)?;
482
483            if encounter.combatants.is_empty() {
484                return Err("No combatants in encounter!".into());
485            }
486
487            encounter.start();
488
489            println!("‚öîÔ∏è Combat started!");
490            println!("\nInitiative order:");
491            for (i, &idx) in encounter.turn_order.iter().enumerate() {
492                let combatant = &encounter.combatants[idx];
493                println!(
494                    "  {}. {} (Initiative: {})",
495                    i + 1,
496                    combatant.name,
497                    combatant.initiative
498                );
499            }
500
501            encounter.save_session(&file)?;
502            println!("\n‚úÖ Encounter saved");
503        }
504
505        CombatCommands::Status { file } => {
506            let encounter = CombatEncounter::load_session(&file)?;
507
508            println!("=== Combat Status ===");
509            println!("Round: {}", encounter.round);
510            println!(
511                "Hope: {}/{}",
512                encounter.hope.current, encounter.hope.maximum
513            );
514            println!("Fear: {}", encounter.fear.current);
515            println!();
516
517            if encounter.is_over() {
518                if let Some(victory) = encounter.player_victory() {
519                    if victory {
520                        println!("üéâ VICTORY! Players won!");
521                    } else {
522                        println!("üíÄ DEFEAT! Enemies won!");
523                    }
524                }
525            } else if encounter.round > 0 {
526                if let Some(current) = encounter.current_combatant() {
527                    println!("Current turn: {}", current.name);
528                    println!();
529                }
530            }
531
532            println!("Combatants:");
533            for combatant in &encounter.combatants {
534                let status = if combatant.is_alive() {
535                    "Alive"
536                } else {
537                    "Dead"
538                };
539                println!(
540                    "  {} [{}] - HP: {}/{}, Evasion: {}, Armor: {}",
541                    combatant.name,
542                    status,
543                    combatant.hp.current,
544                    combatant.hp.maximum,
545                    combatant.evasion,
546                    combatant.armor
547                );
548            }
549        }
550    }
551
552    Ok(())
553}

648fn parse_damage_dice(s: &str) -> Result<DamageDice, Box<dyn std::error::Error>> {
649    // Simple parser for damage dice like "2d6+3" or "1d8+1d4+2"
650    let mut dice = Vec::new();
651    let mut bonus: i16 = 0;
652
653    for part in s.split('+') {
654        let part = part.trim();
655
656        // Check if it's just a number (bonus)
657        if let Ok(n) = part.parse::<i16>() {
658            bonus += n;
659            continue;
660        }
661
662        // Parse dice notation (e.g., "2d6")
663        if let Some((count_str, die_str)) = part.split_once('d') {
664            let count: usize = count_str.parse().unwrap_or(1);
665            let die = parse_die(&format!("d{}", die_str))?;
666
667            for _ in 0..count {
668                dice.push(die);
669            }
670        }
671    }
672
673    Ok(DamageDice::new(dice).with_bonus(bonus))
